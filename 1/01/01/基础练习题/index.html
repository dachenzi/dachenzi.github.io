1、给一个半径，求圆的面积和周长。圆周率3.14

# 简单版本
radius = float(input('>>: ').strip())
print('area: {:.2f},circumference: {:.2f}'.format(3.14*radius**2, 2*3.14*radius))

# 复杂版本
import math
while True:
    radius = input('半径>>>: ').strip()
    if radius.isdecimal():
        radius = float(radius)
        area = math.pi * radius ** 2
        circumference = 2 * math.pi * radius
        print('面积: {:.2f}, 周长：{:.2f}'.format(area, circumference))
    elif radius.lower() == 'q':
        print('\033[31m bye \033[0m')
        break
    else:
        print('input Error ,retry')

2 输入两个数，比较大小后，从小到大升序打印

# 简单版
number1 = int(input('>>>: '))
number2 = int(input('>>>: '))
if number1 < number2:
    print(number1, number2)
else:
    print(number2, number1)

# 负载版
lst = []
for i in range(2):
    lst.append(int(input('{} >>>:'.format(i+1).strip())))
lst.sort()
print('<'.join(map(str, lst)))

# 三元表达式
a = int(input('>>>: '))
b = int(input('>>>: '))
print(a, b) if a > b else print(b, a)

3 依次输入若干个整数，打印出最大值，如果输入为空，则退出程序

# 简单版
max_number = int(input('>>>:'))
while True:
    new_number = int(input('>>>:'))
    if new_number > max_number:
        max_number = new_number
    print(max_number)

# 稍微复杂版
max_number = 0
while True:
    new_number = input('>>>: ').strip()
    if new_number.isdecimal():
        new_number = int(new_number)
        if new_number > max_number:
            max_number = new_number
        print(max_number)
    elif new_number.lower() == 'q':
        print('\033[31mbye\033[0m')
        break
    else:
        print('input Error, retury')

4 输入n个数，求每次输入后的算数平均数

count = 1
while True:
    number = int(input('{} >>>:'.format(count)))
    print('avg: {}'.format(number/count))
    count += 1

5 打印一个边长为n的正方形

side = int(input('len of side >>>: '))
for i in range(side):
    if i == 0 or i == (side - 1):
        print('*'*side)
    else:
        print('*'+ ' '*(side-2) + '*')

7 判断学生成绩，成绩等级A-E,其中90分以上为A,80-89分为B，70-79分为C,60-69分为D，60分以下为E

score = int(input('score: '))
if score >= 90:
    print('A')
elif score >= 80:
    print('B')
elif score >= 70:
    print('C')
elif score >= 60:
    print('D')
else:
    print('E')

# 二分
score = int(input('score: '))
if score >= 70:
    if score < 80:
        print('C')
    elif score < 90:
        print('B')
    else:
        print('A')
else:
    if score >= 60:
        print('D')
    else:
        print('E')

8 求1到5阶乘之和

# 每循环一个数求一次阶乘，然后累加
sum = 0
for i in range(1, 6):
    result = 1
    for j in range(1, i+1):
        result *= j
    sum += result
print(sum)

# 3的阶乘实际上就是2的阶乘结果乘3即可，利用这种规律计算
sum = 1
total = 0
for i in range(1, 6):
    sum *= i
    total += sum
print(total)
