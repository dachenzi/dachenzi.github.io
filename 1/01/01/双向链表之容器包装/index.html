容器包装

  1. 构建双向链表，实现append，pop，insert，remove，iternodes方法
  2. 将双向链表链表封装为容器，并且提供__getitem__、__iter__、__setitem__方法
class LinkNode:
    def __init__(self, value, prev=None, after=None):
        self.value = value
        self.prev = prev
        self.after = after

class LinkedList:
    def __init__(self):
        self.head = None
        self.tail = None
        self.length = 0

    def append(self, value):
        node = LinkNode(value)
        if self.head == None:
            self.head = node
        else:
            prev = self.tail
            prev.after = node
            node.prev = prev
        self.tail = node
        self.length += 1

    def pop(self):
        item = self.tail
        item.prev.after = None
        self.length -= 1
        self.tail = item.prev
        return item.value

    def insert(self, position, value):
        node = LinkNode(value)
        if position < 0:
            raise IndexError('Index must ge Zero')
        elif position == 0:
            if self.head == None and self.tail == None:
                self.head = node
                self.tail = node
            else:
                item = self.head
                self.head = node
                node.after = item
                item.prev = node
            self.length += 1

        elif position < self.length:
            for index,item in enumerate(self._iternode()):
                if index == position:
                    prev = item.prev
                    prev.after = node
                    node.prev = prev
                    node.after = item
                    item.prev = node
                    self.length += 1
        else:
            self.append(value)

    def _modify(self, index, value):
        node = LinkNode(value)
        for i,item in enumerate(self._iternode()):
            if i == index:
                if item.prev == None and item.after == None :
                    print('1')
                    self.head = node
                    self.tail = node
                elif item.prev == None:
                    print(2)
                    print(node.value)
                    item.after.prev = node
                    node.after = item.after
                    self.head = node
                elif item.after == None:
                    print(3)
                    item.prev.prev = node
                    node.prev = item.prev
                    self.tail = node
                else:
                    print(4)
                    item.prev.after = node
                    item.after.prev = node
                    node.prev = item.prev
                    node.after = item.after

    def remove(self, value):
        for node in self._iternode():
            if node.value == value:
                prev = node.prev
                after = node.after
                if prev == None:
                    self.head = after
                    after.prev = None
                elif after == None:
                    self.tail = prev
                    prev.after = None
                else:
                    prev.after = after
                    after.prev = prev
                self.length -= 1
                break
        else:
            raise ValueError('{} is not exists'.format(value))

    def _iternode(self):
        start = self.head
        while start.after :
            yield start
            start = start.after
        yield self.tail

    def __getitem__(self, index):
        if 0 <= index < self.length:
            return self._getitem(index)
        elif 0 <= (self.length + index) < self.length:
            print('index',self.length - 1  + index )
            return self._getitem(self.length + index )
        else:
            raise IndexError('list index out of range')

    def __setitem__(self, index, value):
        if 0 <= index <= self.length:
            return self._modify(index,value)
        elif 0 <= (self.length + index) <= self.length:   # index = -1 , self.length = 1
            return self._modify(self.length + index, value)
        else:
            raise IndexError('list index out of range')

    def _getitem(self,index):
        for i,value in enumerate(self._iternode()):
            if index == i :
                return value.value

    def __len__(self):
        return self.length

    def __iter__(self):
        for node in self._iternode():
            yield node.value
